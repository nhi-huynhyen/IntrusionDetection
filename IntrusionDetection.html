def evaluate_models(X, y, title):
    X_train, X_test, y_train, y_test = split_data(X, y)
    results = {}
    process = psutil.Process(os.getpid())

    # Random Forest
    start_time = time.time()
    start_mem = process.memory_info().rss
    rf = RandomForestClassifier(n_estimators=100, random_state=42)
    rf.fit(X_train, y_train)
    y_pred_rf = rf.predict(X_test)
    end_time = time.time()
    end_mem = process.memory_info().rss
    results['rf_acc'] = accuracy_score(y_test, y_pred_rf)
    results['rf_time'] = end_time - start_time
    results['rf_mem'] = (end_mem - start_mem) / (1024 * 1024)  # in MB

    # XGBoost
    start_time = time.time()
    start_mem = process.memory_info().rss
    xgb = XGBClassifier(n_estimators=100, use_label_encoder=False, eval_metric='mlogloss', random_state=42)
    xgb.fit(X_train, y_train)
    y_pred_xgb = xgb.predict(X_test)
    end_time = time.time()
    end_mem = process.memory_info().rss
    results['xgb_acc'] = accuracy_score(y_test, y_pred_xgb)
    results['xgb_time'] = end_time - start_time
    results['xgb_mem'] = (end_mem - start_mem) / (1024 * 1024)  # in MB

    print(f"\n📊 Results for: {title}")
    print("Random Forest Accuracy:", results['rf_acc'], "| Time:", round(results['rf_time'], 2), "s | Mem:", round(results['rf_mem'], 2), "MB")
    print("XGBoost Accuracy:", results['xgb_acc'], "| Time:", round(results['xgb_time'], 2), "s | Mem:", round(results['xgb_mem'], 2), "MB")
    return results

# VAE
def generate_with_vae(X_train):
    input_dim = X_train.shape[1]
    latent_dim = 10

    inputs = Input(shape=(input_dim,))
    h = Dense(128, activation='relu')(inputs)
    z_mean = Dense(latent_dim)(h)
    z_log_var = Dense(latent_dim)(h)

    def sampling(args):
        z_mean, z_log_var = args
        epsilon = tf.random.normal(shape=(tf.shape(z_mean)[0], latent_dim))
        return z_mean + tf.exp(0.5 * z_log_var) * epsilon

    z = Lambda(sampling)([z_mean, z_log_var])
    decoder_h = Dense(128, activation='relu')
    decoder_out = Dense(input_dim, activation='sigmoid')
    outputs = decoder_out(decoder_h(z))

    vae = Model(inputs, outputs)
    vae.compile(optimizer=Adam(0.001), loss='mse')
    vae.fit(X_train, X_train, epochs=50, batch_size=32, verbose=0)

    encoder = Model(inputs, z_mean)
    decoder_input = Input(shape=(latent_dim,))
    decoder_model = Model(decoder_input, decoder_out(decoder_h(decoder_input)))

    z_samples = np.random.normal(size=(len(X_train), latent_dim))
    X_generated = decoder_model.predict(z_samples)
    return X_generated

# CGAN
def generate_with_cgan(X_train, y_train, num_samples=None, epochs=3000):
    latent_dim = 100
    num_classes = len(np.unique(y_train))
    input_dim = X_train.shape[1]
    if num_samples is None:
        num_samples = len(X_train)

    def build_generator():
        noise = Input(shape=(latent_dim,))
        label = Input(shape=(1,), dtype='int32')
        label_embedding = Flatten()(Embedding(num_classes, latent_dim)(label))
        model_input = tf.keras.layers.multiply([noise, label_embedding])
        x = Dense(128)(model_input)
        x = tf.keras.layers.LeakyReLU(0.2)(x)
        x = Dense(input_dim, activation='sigmoid')(x)
        return Model([noise, label], x)

    def build_discriminator():
        data_input = Input(shape=(input_dim,))
        label = Input(shape=(1,), dtype='int32')
        label_embedding = Flatten()(Embedding(num_classes, input_dim)(label))
        model_input = Concatenate()([data_input, label_embedding])
        x = Dense(128)(model_input)
        x = tf.keras.layers.LeakyReLU(0.2)(x)
        x = Dense(1, activation='sigmoid')(x)
        return Model([data_input, label], x)

    generator = build_generator()
    discriminator = build_discriminator()
    discriminator.compile(loss='binary_crossentropy', optimizer=Adam(0.0002, 0.5), metrics=['accuracy'])

    noise_input = Input(shape=(latent_dim,))
    label_input = Input(shape=(1,))
    generated_sample = generator([noise_input, label_input])
    discriminator.trainable = False
    validity = discriminator([generated_sample, label_input])
    combined = Model([noise_input, label_input], validity)
    combined.compile(loss='binary_crossentropy', optimizer=Adam(0.0002, 0.5))

    for epoch in range(epochs):
        idx = np.random.randint(0, X_train.shape[0], 32)
        real_samples = X_train[idx]
        real_labels = y_train[idx]
        noise = np.random.normal(0, 1, (32, latent_dim))
        gen_labels = np.random.randint(0, num_classes, 32)
        gen_samples = generator.predict([noise, gen_labels], verbose=0)

        d_loss_real = discriminator.train_on_batch([real_samples, real_labels], np.ones((32, 1)))
        d_loss_fake = discriminator.train_on_batch([gen_samples, gen_labels], np.zeros((32, 1)))
        d_loss = 0.5 * np.add(d_loss_real, d_loss_fake)

        noise = np.random.normal(0, 1, (32, latent_dim))
        sampled_labels = np.random.randint(0, num_classes, 32)
        g_loss = combined.train_on_batch([noise, sampled_labels], np.ones((32, 1)))

    noise = np.random.normal(0, 1, (num_samples, latent_dim))
    sampled_labels = np.random.randint(0, num_classes, num_samples)
    generated_samples = generator.predict([noise, sampled_labels], verbose=0)
    return generated_samples, sampled_labels
